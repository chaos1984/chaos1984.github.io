0.基本排列组合算法

import numpy as np

def A_n_m(n,m):
    return float(np.math.factorial(n))/float(np.math.factorial(n-m))

def C_n_m(n,m):
    return  A_n_m(n,m)/np.math.factorial(m)

def Prob(t,n,m):
    return  C_n_m(n,m)*C_n_m(t-n,n-m)/C_n_m(t,n)
